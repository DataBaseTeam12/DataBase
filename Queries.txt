*****MEDIA*****

//DISPLAY ALL MEDIA
SELECT title, first_name, last_name, genre, publisher, 
	published_date, audience, language
FROM Author_Media_View;

//SELECT MEDIA BY Author
SELECT Media.title, Author.first_name, Author.last_name, 
	Media.genre, Media.publisher, Media.published_date,
	Media.audience, Media.language
FROM Media
RIGHT JOIN Author ON Media.id = Author.media_id
WHERE Author.first_name = 'first name' 
AND Author.last_name = 'last name';

	published_date, audience, language, edition, type, num_pages
FROM Full_Book_View;

TYPE , num_pages
FROM Full_Book_View
WHERE title =  'title';

TYPE , num_pages
FROM Full_Book_View
WHERE genre = 'genre';

TYPE , num_pages
FROM Full_Book_View

*****MEMBERS*****

//DISPLAY ALL MEMBERS BY IN ALPHABETICAL  ORDER
SELECT *
FROM Member
ORDER BY last_name ASC;

//DISPLAY MEMBERS BY NAME WITH ID IN ASCENDING ORDER
SELECT *
FROM Member
WHERE first_name = 'first_name' AND last_name = 'last_name'
ORDER BY id ASC;

//DISPLAY MEMBER BY ID
SELECT *
FROM Member
WHERE id = id;

//DISPLAY MEMBER BY USERNAME
SELECT *
FROM Member
WHERE username = 'username';

//DISPLAY MEMBERS BY TOTAL FINES IN DESCENDING ORDER
SELECT *
FROM Member
ORDER BY total_fines DESC;
